# 带有记忆功能的函数

const memoizer = (memo, formula) => {
  const recur = n => {
    let result = memo[n]
    if (typeof result !== `number`) {
      memo[n] = result = formula(recur, n)
    }
    return result
  }
  return recur
}

const fibonacci = memoizer([0, 1], (recur, n) => {
  return recur(n - 1) + recur(n - 2)
})

console.time()
for (let i = 0; i <= 10; i++) {
  console.log(fibonacci(i))
}
console.timeEnd()
